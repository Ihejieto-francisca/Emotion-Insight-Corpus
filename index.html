# Example: Top words for one emotion (e.g., "joy")
from sklearn.feature_extraction.text import CountVectorizer

# Filter data for "joy"
joy_comments = df[df['Emotion'] == 'joy']['Comment']

# Get top 10 words
vectorizer = CountVectorizer(stop_words='english')
X = vectorizer.fit_transform(joy_comments)
word_counts = pd.DataFrame(X.sum(axis=0), columns=vectorizer.get_feature_names_out()).T
word_counts.columns = ['Count']
top_words = word_counts.sort_values('Count', ascending=False).head(10)

plt.figure(figsize=(10, 5))
plt.barh(top_words.index, top_words['Count'], color='#2196F3')
plt.title('Top Words Associated with "Joy"')
plt.xlabel('Frequency')
plt.gca().invert_yaxis()
plt.savefig("linguistic feature.png")
plt.show()

# Step 1: Define sensitive keywords (e.g., mental health-related)
sensitive_keywords = ['depression', 'anxiety', 'suicide', 'stress']

# Step 2: Filter comments containing these keywords (case-insensitive)
df['Contains_Sensitive'] = df['Comment'].str.lower().str.contains('|'.join(sensitive_keywords))
sensitive_df = df[df['Contains_Sensitive']]

# Step 3: Calculate emotion distribution for sensitive comments
if not sensitive_df.empty:
    emotion_counts = sensitive_df['Emotion'].value_counts()
else:
    # Hypothetical fallback data (if no matches)
    emotion_counts = pd.Series({
        'sadness': 45,
        'anger': 30,
        'fear': 20,
        'joy': 5
    })

# Step 4: Create a bar chart
plt.figure(figsize=(10, 5))
